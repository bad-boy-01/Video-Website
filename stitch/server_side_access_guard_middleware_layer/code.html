async function checkAccess(req, res) {
  // 1. Get Auth Context
  const user = req.user;
  const { videoId } = req.params;

  // 2. Query Firestore Entitlements
  const hasAccess = await db
    .collection('users')
    .doc(user.uid)
    .checkEntitlement(videoId);

  // 3. Decision Logic
  if (!hasAccess) {
    return res.status(403)
      .json({ error: 'Forbidden' });
  }

  // 4. Generate Signed Handshake
  const url = await storage
    .getSignedUrl(videoId);
    
  res.redirect(url);
}
